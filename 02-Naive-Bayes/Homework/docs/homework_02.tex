    \section{MLE}
        Consider the following very simple model for stock pricing. The price at the end of each day is the price of the previous day multiplied by a fixed, but unknown, rate of return, $\alpha$, with some noise, $w$. For a two-day period, we can observe the following sequence
        \begin{align*}
            y_{2} &= \alpha y_{1} + w_{1} \\
            y_{1} &= \alpha y_{0} + w_{0}
        \end{align*}

        where the noises $w_0, w_1$ are iid with the distribution $N(0, \sigma^2)$, $y_0 \sim N(0, \lambda)$ is independent of the noise sequence. $\sigma^2$ and $\lambda$ are known, while $\alpha$ is unknown.

        \subsection{Problem T1}
            Find the MLE of the rate of return, $\alpha$, given the observed price at the end of each day $y_2, y_1, y_0$. In other words, compute for the value of $\alpha$ that maximizes $p(y_2, y_1, y_0|\alpha)$

            \vspace{1em}

            \textbf{Hint:} This is a Markov process, e.g. $y_2$ is independent of $y_0$ given $y_1$. In general, a process is Markov if $p(y_n|y_{n-1}, y_{n-2}, \dots) = p(y_n|y_{n-1})$. In other words, the present is independent of the past $(y_{n-2}, y_{n-3}, \dots)$, conditioned on the immediate past $y_{n-1}$. You may also find the steps of the proof for logistic regression we did in class useful.

            \vspace{1em}

            \sol In the problem, the goal is to maximizes the joint probability density function (PDF)
            \begin{align*}
                p(y_{2}, y_{1}, y_{0}|\alpha)
            \end{align*}

            \noindent According to the problem, it is given that:
            \begin{itemize}
                \item $y_{0} \sim N(0, \lambda)$
                \item $y_{1} = \alpha y_{0} + w_{0}$ where $w_{0} \sim N(0, \sigma^{2})$
                \item $y_{2} = \alpha y_{1} + w_{1}$ where $w_{1} \sim N(0, \sigma^{2})$
            \end{itemize}

            Using the definition of the conditional probability, we rewrite the equation as:
            \begin{align*}
                p(y_{2}, y_{1}, y_{0}|\alpha) = p(y_{2}|y_{1}, y_{0}, \alpha) \cdot p(y_{1}|y_{0}, \alpha) \cdot p(y_{0}|\alpha)
            \end{align*}

            Since the distribution of $y_{0}$ is given that $N(0, \lambda)$ which is not depends on $\alpha$, we can simplify $p(y_{0}|\alpha)$ to $p(y_{0})$
            \begin{align*}
                p(y_{2}, y_{1}, y_{0}|\alpha) = p(y_{2}|y_{1}, y_{0}, \alpha) \cdot p(y_{1}|y_{0}, \alpha) \cdot p(y_{0})
            \end{align*}

            According to the hint, it is said that in the Markov process, the term $p(y_{2}|y_{1}, y_{0} \alpha)$ can be simplified to $p(y_{2}|y_{1}, \alpha)$
            \begin{align*}
                p(y_{2}, y_{1}, y_{0}|\alpha) = p(y_{2}|y_{1}, \alpha) \cdot p(y_{1}|y_{0}, \alpha) \cdot p(y_{0})
            \end{align*}

            Now, the Likelihood function $L(\alpha)$ becomes:
            \begin{align*}
                L(\alpha) = p(y_{2}|y_{1}, \alpha) \cdot p(y_{1}|y_{0}, \alpha) \cdot p(y_{0})
            \end{align*}

    \pagebreak

            \noindent\underline{\textbf{Probability Density Function}}

            \vspace{1em}

            From probability density function (PDF) of normal distribution $N(\mu, \sigma^{2})$ is:
            \begin{align*}
                f(x) = \frac{1}{\sqrt{2\pi\sigma^{2}}} \exp\left( -\frac{(x - \mu)^{2}}{2\sigma^{2}} \right)
            \end{align*}

            From $p(y_{1}|y_{0}, \alpha)$ which already given $y_{0}$ value and equation $y_{1} = \alpha y_{0} + w_{0}$ where \\[0.5em]
            $w_{0} \sim N(0, \sigma^{2})$, we can say that:
            \begin{itemize}
                \item $y_{1}$ is a sum of a constant ($\alpha y_{0}$) and a random variable $w_{0} \sim N(0, \sigma^{2})$.
                \item \textbf{Mean}: $\mu = E[y_{1}] = \alpha y_{0}$
                \item \textbf{Variance}: $Var(y_{1}) = \sigma^{2}$
            \end{itemize}

            So the probability density function is:
            \begin{align*}
                p(y_{1}|y_{0}, \alpha) = \frac{1}{\sqrt{2\pi\sigma^{2}}} \exp\left( -\frac{(y_{1} - \alpha y_{0})^{2}}{2\sigma^{2}} \right)
            \end{align*}

            \vspace{1em}

            From $p(y_{2}|y_{1}, \alpha)$ which already given $y_{1}$ value and equation $y_{2} = \alpha y_{1} + w_{1}$ where \\[0.5em]
            $w_{1} \sim N(0, \sigma^{2})$, we can say that:
            \begin{itemize}
                \item $y_{2}$ is a sum of constant ($\alpha y_{1}$) and random variable $w_{1} \sim N(0, \sigma^{2})$.
                \item \textbf{Mean}: $\mu = E[y_{2}] = \alpha y_{1}$
                \item \textbf{Variance}: $Var(y_{2}) = \sigma^{2}$
            \end{itemize}

            So the probability density function is:
            \begin{align*}
                p(y_{2}|y_{1}, \alpha) = \frac{1}{\sqrt{2\pi\sigma^{2}}} \exp\left( -\frac{(y_{2} - \alpha y_{1})^{2}}{2\sigma^{2}} \right)
            \end{align*}

    \pagebreak

            \noindent\underline{\textbf{Log-Likelihood}}
            
            \vspace{1em}
            
            \noindent Apply the natural logarithm function into both sides of the equations.
            \begin{align*}
                L(\alpha) &= p(y_{2}|y_{1}, \alpha) \cdot p(y_{1}|y_{0}, \alpha) \cdot p(y_{0}) \\[1em]
                \ln(L(\alpha)) &= \ln(p(y_{2}|y_{1}, \alpha) \cdot p(y_{1}|y_{0}, \alpha) \cdot p(y_{0})) \\[1em]
                \ln(L(\alpha)) &= \ln(p(y_{2}|y_{1}, \alpha)) + \ln(p(y_{1}|y_{0}, \alpha)) + \ln(p(y_{0})) \\[1em]
                \ln(L(\alpha)) &= \ln\left[
                    \frac{1}{\sqrt{2\pi\sigma^{2}}} \exp\left( -\frac{(y_{1} - \alpha y_{0})^{2}}{2\sigma^{2}} \right)
                    \right]
                    + \ln\left[
                        \frac{1}{\sqrt{2\pi\sigma^{2}}} \exp\left( -\frac{(y_{2} - \alpha y_{1})^{2}}{2\sigma^{2}} \right)
                    \right]
                    + \ln(p(y_{0})) \\[1em]
                \ln(L(\alpha)) &= \ln\left[
                    \exp\left( -\frac{(y_{1} - \alpha y_{0})^{2}}{2\sigma^{2}} \right)
                    \right]
                    + \ln\left[
                        \exp\left( -\frac{(y_{2} - \alpha y_{1})^{2}}{2\sigma^{2}} \right)
                    \right]
                    + \left[
                        2 \cdot \ln\left( \frac{1}{\sqrt{2\pi\sigma^{2}}} \right)
                        + \ln(p(y_{0}))
                    \right] \\[1em]
                \ln(L(\alpha)) &= -\frac{(y_{1} - \alpha y_{0})^{2}}{2\sigma^{2}}
                    -\frac{(y_{2} - \alpha y_{1})^{2}}{2\sigma^{2}}
                    + \left[
                        2 \cdot \ln\left( \frac{1}{\sqrt{2\pi\sigma^{2}}} \right)
                        + \ln(p(y_{0}))
                    \right] \\[1em]
                \ln(L(\alpha)) &= -\frac{1}{2\sigma^{2}} \left[  
                        (y_{1} - \alpha y_{0})^{2} + (y_{2} - \alpha y_{1})^{2}
                    \right]
                    + \left[
                        2 \cdot \ln\left( \frac{1}{\sqrt{2\pi\sigma^{2}}} \right)
                        + \ln(p(y_{0}))
                    \right]
            \end{align*}

            \vspace{1em}

            Then, calculate $\alpha$ that maximizes log-likelihood function. Local maximum point of the log-likelihood function can be solved by the equation below.
            \begin{align*}
                \frac{d}{d\alpha}\left[ \ln(L(\alpha)) \right] &= 0
            \end{align*}

            We are using the simplified log-likelihood function from the previous section.
            \begin{align*}
                \frac{d}{d\alpha}\left[
                    -\frac{1}{2\sigma^{2}} \left[
                        (y_{1} - \alpha y_{0})^{2} + (y_{2} - \alpha y_{1})^{2}
                    \right]
                    + \left[
                        2 \cdot \ln\left( \frac{1}{\sqrt{2\pi\sigma^{2}}} \right)
                        + \ln(p(y_{0}))
                    \right]
                \right] &= 0
            \end{align*}

            The equation continues in the next page $\rightarrow$

            The last term is a constant which its derivative is 0.
            \begin{align*}
                \frac{d}{d\alpha}\left[
                    -\frac{1}{2\sigma^{2}} \left[
                        (y_{1} - \alpha y_{0})^{2} + (y_{2} - \alpha y_{1})^{2}
                    \right]
                \right] &= 0 \\[1em]
                -\frac{1}{2\sigma^{2}} \left[
                    \frac{d}{d\alpha} (y_{1} - \alpha y_{0})^{2}
                    + \frac{d}{d\alpha} (y_{2} - \alpha y_{1})^{2}
                \right] &= 0 \\[1em]
                \frac{d}{d\alpha} (y_{1} - \alpha y_{0})^{2}
                + \frac{d}{d\alpha} (y_{2} - \alpha y_{1})^{2} &= 0 \\[1em]
                2(y_{1} - \alpha y_{0}) \cdot \frac{d}{d\alpha}(y_{1} - \alpha y_{0})
                + 2(y_{2} - \alpha y_{1}) \cdot \frac{d}{d\alpha}(y_{2} - \alpha y_{1}) &= 0 \\[1em]
                -2y_{0}(y_{1} - \alpha y_{0}) - 2y_{1}(y_{2} - \alpha y_{1}) &= 0 \\[1em]
                y_{0}(y_{1} - \alpha y_{0}) + y_{1}(y_{2} - \alpha y_{1}) &= 0 \\[1em]
                (y_{0}y_{1} - \alpha {y_{0}}^{2}) + (y_{1}y_{2} - \alpha {y_{1}}^{2}) &= 0 \\[1em]
                y_{0}y_{1} + y_{1}y_{2} - \alpha ({y_{0}}^{2} - {y_{1}}^{2}) &= 0 \\[1em]
                - \alpha ({y_{0}}^{2} + {y_{1}}^{2}) &= - y_{0}y_{1} - y_{1}y_{2} \\[1em]
                \alpha ({y_{0}}^{2} + {y_{1}}^{2}) &= y_{0}y_{1} + y_{1}y_{2} \\[1em]
                \alpha &= \frac{y_{0}y_{1} + y_{1}y_{2}}{{y_{0}}^{2} + {y_{1}}^{2}}
            \end{align*}

            \ans The Maximum Likelihood Estimator (MLE) is
            \begin{align*}
                {\hat{\alpha}}_{MLE} = \frac{y_{0}y_{1} + y_{1}y_{2}}{{y_{0}}^{2} + {y_{1}}^{2}}
            \end{align*}

    \section{Simple Bayes Classifier}

        A student in Pattern Recognition course had finally built the ultimate classifier for cat emotions. He used one input features: the amount of food the cat ate that day, $x$ (Being a good student he already normalized $x$ to standard Normal). He proposed the following likelihood probabilities for class 1 (happy cat) and 2 (sad cat)
        \begin{align*}
            P(x|w_1) &= N(4, 2) \\
            P(x|w_2) &= N(0, 2)
        \end{align*}

        \subsection{Problem T2}
            Plot the posteriors values of the two classes on the same axis. Using the likelihood ratio test, what is the decision boundary for this classifier? Assume equal prior probabilities.

            \vspace{1em}

            \sol First, we are going to dealing with probability density function (PDF) first.
            \vspace{1em}

            \noindent\underline{\textbf{Probability Density Function}}
            \vspace{0.5em}

            From probability density function (PDF) of normal distribution $N(\mu, \sigma^{2})$ is:
            \begin{align*}
                f(x) = \frac{1}{\sqrt{2\pi\sigma^{2}}} \exp\left( -\frac{(x - \mu)^{2}}{2\sigma^{2}} \right)
            \end{align*}
            
            According to the the problem, the probability density function of these two classes are as follows:
            \begin{align*}
                P(x|w_1) &= N(4, 2) \rightarrow \mu = 4, {\sigma}^{2} = 2 \\
                P(x|w_2) &= N(0, 2) \rightarrow \mu = 0, {\sigma}^{2} = 2
            \end{align*}
            \begin{align*}
                P(x|w_1) &= \frac{1}{2\sqrt{\pi}} \exp\left( -\frac{(x - 4)^{2}}{4} \right) \\[1em]
                P(x|w_2) &= \frac{1}{2\sqrt{\pi}} \exp\left( -\frac{{x}^{2}}{4} \right)
            \end{align*}

    \pagebreak

            \noindent\underline{\textbf{Bayes Theorem}}
            \vspace{0.5em}

            According to the Bayes Theorem, the posterior probability can be written as following equation:
            \begin{align*}
                P(w_i|x) &= \frac{P(x|w_i)P(w_i)}{P(x)}
            \end{align*}
            \begin{itemize}
                \item $P(w_i|x)$ is Posterior Probability which means the probability of $w_i$ given $x$.
                \item $P(x|w_i)$ is Likelihood which means the probability of $x$ given $w_i$.
                \item $P(w_i)$ is Prior Probability which means the probability of $w_i$.
                \item $P(x)$ is Evidence which means the probability of $x$.
            \end{itemize}

            We are considering decision boundary where both posterior probabilities are equal.
            \begin{align*}
                P(w_1|x) &= P(w_2|x) \\[1em]
                \frac{P(x|w_1)P(w_1)}{P(x)} &= \frac{P(x|w_2)P(w_2)}{P(x)} \\[1em]
                P(x|w_1)P(w_1) &= P(x|w_2)P(w_2)
            \end{align*}
			
			\vspace{0.5em}

            Assume both two classes have equal priors; therefore, $P(w_1) = 0.5$ and $P(w_2) = 0.5$.
            \begin{align*}
                P(x|w_1) \cdot 0.5 &= P(x|w_2) \cdot 0.5 \\
                P(x|w_1) &= P(x|w_2) \\[1em]
                \frac{1}{2\sqrt{\pi}} \exp\left( -\frac{(x - 4)^{2}}{4} \right) &= \frac{1}{2\sqrt{\pi}} \exp\left( -\frac{{x}^{2}}{4} \right) \\[1em]
                \exp\left( -\frac{(x - 4)^{2}}{4} \right) &= \exp\left( -\frac{{x}^{2}}{4} \right) \\[1em]
                -\frac{(x - 4)^{2}}{4} &= -\frac{{x}^{2}}{4} \\[1em]
                (x - 4)^{2} &= {x}^{2} \\
                (x - 4)^{2} - {x}^{2} &= 0 \\
                ((x - 4) - x)((x - 4) + x) &= 0 \\
                (-4)(2x - 4) &= 0 \\
                x &= 2
            \end{align*}

    \pagebreak

            \noindent\underline{\textbf{Plotting Posterior Probability}}
            \vspace{0.5em}

            We are going to use \texttt{matplotlib} for plotting posterior probability of both classes, so these imports are mandatory.
            \begin{codeblock}{Dependencies}
                import numpy as np
                import matplotlib.pyplot as plt
                from scipy.stats import norm
            \end{codeblock}

            Define $\mu$ and $\sigma$ of both $P(x|w_1)$ and $P(x|w_2)$, also the $x$ variable range for graph visualization.
            \begin{codeblock}{Define Mean and Variance of the Distrbution}
                # Parameters
                MEAN_1, VAR_1 = 4, 2
                MEAN_2, VAR_2 = 0, 2
                SIGMA = np.sqrt(VAR_1)

                # Range for x
                X = np.linspace(-4, 8, 1000)
            \end{codeblock}

            Define prior probabilities and likelihood probabilities of both classes.
            \begin{codeblock}{Define Prior and Likelihood probabilities}
                # Likelihoods P(x|w)
                likelihood_1 = norm.pdf(X, MEAN_1, SIGMA)
                likelihood_2 = norm.pdf(X, MEAN_2, SIGMA)

                # Priors
                PRIOR_1 = 0.5
                PRIOR_2 = 0.5
            \end{codeblock}

            Calculate Evidence probability which can be calculated as following equation.
            \begin{align*}
                P(x) = P(x|w_1)P(w_1) + P(x|w_2)P(w_2)
            \end{align*}
            \begin{codeblock}{Calculate Evidence Probability}
                # Evidence P(x)
                evidence = (likelihood_1 * PRIOR_1) + (likelihood_2 * PRIOR_2)
            \end{codeblock}

            Calculate Posterior probability of both classes by using Bayes Theorem.
            \begin{align*}
                P(w_i|x) &= \frac{P(x|w_i)P(w_i)}{P(x)}
            \end{align*}
            \begin{codeblock}{Calculate Posterior Probabilities}
                # Posteriors P(w|x)
                posterior1 = (likelihood_1 * PRIOR_1) / evidence
                posterior2 = (likelihood_2 * PRIOR_2) / evidence
            \end{codeblock}

    \pagebreak

            Visualize posterior probability of both classes with decision boundary line by using \texttt{matplotlib} library.
            \begin{codeblock}{Posterior Probabilities Visualization}
                # Plot
                plt.figure(figsize=(10, 6))
                plt.plot(X, posterior1, label=r"Happy Cat $P(w_1|x)$", color="blue", linewidth=2)
                plt.plot(X, posterior2, label=r"Sad Cat $P(w_2|x)$", color="red", linewidth=2)

                # Decision Boundary (Intersection)
                plt.axvline(x=2, color="green", linestyle="--", label="Decision Boundary (x=2)")

                # Plot Details
                plt.title("Posterior Probabilities (Problem T2)", fontsize=14)
                plt.xlabel("Normalized Food Amount (x)", fontsize=12)
                plt.ylabel("Probability", fontsize=12)
                plt.legend(fontsize=12)
                plt.grid(True, alpha=0.3)
                plt.tight_layout()

                # Save plot
                plt.savefig("figure_01.png")
            \end{codeblock}

            \vspace{1em}

            \ans The decision boundary for this classifier is at $x = 2$ which can be seen from the visualization below.

                \caption{Posterior Probabilities with Decision Boundary}
            \end{figure}

        \subsection{Problem T3}
            What happen to the decision boundary if the cat is happy with a prior of 0.75?
            \vspace{1em}

            \sol To find a decision boundary, we need to find a point that both posterior probability of two classes are equal.
            \begin{align*}
                P(w_1|x) &= P(w_2|x) \\[1em]
                \frac{P(x|w_1)P(w_1)}{P(x)} &= \frac{P(x|w_2)P(w_2)}{P(x)} \\[1em]
                P(x|w_1)P(w_1) &= P(x|w_2)P(w_2)
            \end{align*}

            The prior of happy cat $P(w_1)$ is 0.75, so the prior of sad cat $P(w_2)$ is 1 - 0.75 = 0.25.
            \begin{align*}
                P(x|w_1) \cdot 0.75 &= P(x|w_2) \cdot 0.25 \\
                3 \cdot P(x|w_1) &= P(x|w_2) \\[1em]
                \frac{3}{2\sqrt{\pi}} \exp\left( -\frac{(x - 4)^{2}}{4} \right) &= \frac{1}{2\sqrt{\pi}} \exp\left( -\frac{{x}^{2}}{4} \right) \\[1em]
                3 \cdot \exp\left( -\frac{(x - 4)^{2}}{4} \right) &= \exp\left( -\frac{{x}^{2}}{4} \right) \\[1em]
                3 \cdot \frac{\exp\left( -\frac{(x - 4)^{2}}{4} \right)}{\exp\left( -\frac{{x}^{2}}{4} \right)} &= 1 \\[1em]
                \exp\left[ \left( -\frac{(x - 4)^{2}}{4} \right) - \left( -\frac{{x}^{2}}{4} \right) \right] &= \frac{1}{3} \\[1em]
                \exp\left( \frac{{x}^{2} - (x - 4)^{2}}{4} \right) &= \frac{1}{3} \\[1em]
                \frac{{x}^{2} - (x - 4)^{2}}{4} &= \ln\left( \frac{1}{3} \right) \\[1em]
                \frac{(x - (x - 4))(x + (x - 4))}{4} &= -\ln(3) \\
                2x - 4 &= -\ln(3) \\
                x &= \frac{4 -\ln(3)}{2}
            \end{align*}

            \ans The decision boundary for this classifier is at $x \approx 1.45$.
